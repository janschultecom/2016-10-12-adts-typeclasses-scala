import org.specs2.mutable.Spec

class DaySpec extends Spec{

  "Sunday" should {
    "be greater than Monday" in {

      Day.SUNDAY must be_>(Day.MONDAY)
    }
  }

  "Monday" should {
    "be smaller than Sunday" in {
      Day.MONDAY must be_<(Day.SUNDAY)
    }
  }
}
